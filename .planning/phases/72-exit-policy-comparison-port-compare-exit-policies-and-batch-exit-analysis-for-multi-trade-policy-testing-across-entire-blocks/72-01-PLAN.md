```typescript
import type { TriggerType, ExitTriggerConfig, TriggerFireEvent, LegGroupConfig } from './exit-triggers.js';
import type { PnlPoint, ReplayLeg } from './trade-replay.js';

export type BaselineMode = 'actual' | 'holdToEnd';

export interface BatchExitConfig {
  candidatePolicy: ExitTriggerConfig[];      // Triggers to evaluate
  legGroups?: LegGroupConfig[];              // Optional per-group triggers
  baselineMode: BaselineMode;                // 'actual' or 'holdToEnd'
  format: 'summary' | 'full';               // Output density
}

export interface TradeInput {
  tradeIndex: number;
  dateOpened: string;                        // YYYY-MM-DD
  actualPnl: number;                        // From tradelog pl field
  pnlPath: PnlPoint[];                      // Full replay path
  legs: ReplayLeg[];                         // Replay legs
}

export interface TradeExitResult {
  tradeIndex: number;
  dateOpened: string;
  actualPnl: number;                        // Actual P&L from tradelog
  candidatePnl: number;                     // P&L if candidate policy was used
  baselinePnl: number;                      // Baseline P&L (actual or holdToEnd)
  pnlDelta: number;                         // candidatePnl - baselinePnl
  triggerFired: TriggerType | 'noTrigger';  // Which trigger fired first
  fireTimestamp: string | null;              // When it fired
}

export interface TriggerAttribution {
  trigger: TriggerType | 'noTrigger';
  count: number;                            // How many trades this trigger fired first
  avgPnl: number;                           // Avg candidate P&L when this trigger fired
  totalPnl: number;                         // Sum candidate P&L
  avgDelta: number;                         // Avg pnlDelta vs baseline
}

export interface AggregateStats {
  totalTrades: number;
  winningTrades: number;                    // candidatePnl > 0
  losingTrades: number;                     // candidatePnl < 0
  winRate: number;                          // winningTrades / totalTrades
  totalPnl: number;                         // Sum of candidatePnl
  avgPnl: number;                           // Mean candidatePnl
  avgWin: number;                           // Mean of winning candidatePnls
  avgLoss: number;                          // Mean of losing candidatePnls
  maxWin: number;
  maxLoss: number;
  profitFactor: number;                     // sum(wins) / abs(sum(losses)), Infinity if no losses
  maxDrawdown: number;                      // Max sequential drawdown from equity curve
  sharpeRatio: number | null;               // mean/stddev of candidatePnls, null if < 2 trades
  maxWinStreak: number;
  maxLossStreak: number;
  // Deltas vs baseline
  baselineTotalPnl: number;
  totalPnlDelta: number;                    // totalPnl - baselineTotalPnl
  baselineWinRate: number;
}

export interface BatchExitResult {
  aggregate: AggregateStats;
  triggerAttribution: TriggerAttribution[];
  perTrade: TradeExitResult[];              // Empty if format="summary"
  baselineMode: BaselineMode;
  summary: string;
  profileContext?: {
    structureType: string;
    exitRules: string[];
  };
}
```

**Functions to implement:**

1. `computeAggregateStats(tradeResults: TradeExitResult[]): AggregateStats`
   - Win rate: count(candidatePnl > 0) / total
   - Profit factor: sum(positive candidatePnls) / abs(sum(negative candidatePnls)). If no losses, return Infinity.
   - Max drawdown: Build equity curve from sequential candidatePnls (cumsum), compute max peak-to-trough.
   - Sharpe: mean(candidatePnls) / stddev(candidatePnls). Use sample stddev (N-1). Return null if < 2 trades.
   - Streaks: Count consecutive wins/losses.
   - Baseline aggregates: same calculations on baselinePnl.

2. `computeTriggerAttribution(tradeResults: TradeExitResult[]): TriggerAttribution[]`
   - Group by triggerFired. For each group: count, avg candidatePnl, total candidatePnl, avg pnlDelta.
   - Sort by count descending.
   - Include 'noTrigger' category for trades where no trigger fired.

3. `analyzeBatch(trades: TradeInput[], config: BatchExitConfig): BatchExitResult`
   - For each trade:
     a. Import and call `analyzeExitTriggers` from `exit-triggers.ts` with the trade's pnlPath, legs, and config.candidatePolicy (plus config.legGroups).
     b. Determine candidatePnl: if firstToFire exists, use firstToFire.pnlAtFire. If no trigger fires, use last pnlPath point's strategyPnl (hold to end of replay).
     c. Determine baselinePnl: if baselineMode="actual", use trade.actualPnl. If baselineMode="holdToEnd", use last pnlPath point's strategyPnl.
     d. Compute pnlDelta = candidatePnl - baselinePnl.
     e. Build TradeExitResult.
   - Call computeAggregateStats and computeTriggerAttribution.
   - Build summary string: "Analyzed N trades: candidate win rate X%, total P&L $Y (delta $Z vs baseline). Top trigger: {type} fired on M trades."
   - If format="summary", set perTrade to empty array.
   - Return BatchExitResult.

**Important:** This module has NO imports from DuckDB, fetch, or massive-client. It only imports from `exit-triggers.ts` and `trade-replay.ts` for types.
