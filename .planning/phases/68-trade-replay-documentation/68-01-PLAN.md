    ```typescript
    export interface ReplayLeg {
      occTicker: string;       // Full OCC ticker for Massive API fetch
      quantity: number;        // Positive = long, negative = short
      entryPrice: number;      // Per-contract entry price
      multiplier: number;      // 100 for standard equity/index options
    }

    export interface PnlPoint {
      timestamp: string;       // "YYYY-MM-DD HH:MM" ET
      strategyPnl: number;     // Combined P&L across all legs at this minute
      legPrices: number[];     // HL2 mark price for each leg at this minute
    }

    export interface ReplayResult {
      pnlPath: PnlPoint[];
      mfe: number;             // Max of strategyPnl series (per D-09)
      mae: number;             // Min of strategyPnl series (per D-09)
      mfeTimestamp: string;    // When MFE occurred
      maeTimestamp: string;    // When MAE occurred
      totalPnl: number;       // Final P&L at last bar
      legs: ReplayLeg[];      // The legs that were replayed
    }

    export interface ParsedLeg {
      root: string;            // "SPY", "SPX", "SPXW"
      strike: number;          // Numeric strike price
      type: "C" | "P";        // Call or Put
      quantity: number;        // +1 or -1 (direction derived from position in spread)
    }
    ```

    **Functions:**

    1. `parseLegsString(legsStr: string): ParsedLeg[]`
       - Parse formats: "SPY 470C", "SPY 470C/465C", "SPX 4500P/4450P"
       - Split on "/" delimiter for multi-leg
       - Regex for each leg: `/^([A-Z]+)\s+(\d+(?:\.\d+)?)(C|P)$/i` for compact format
       - Also handle verbose: "SPY Jan25 470 Call" → extract root, strike, type
       - First leg gets quantity=+1 (bought), subsequent legs alternate -1, +1, -1 (standard spread convention)
       - If no strikes can be extracted, throw Error with message: `Cannot parse legs "${legsStr}" — use hypothetical mode with explicit strikes`

    2. `buildOccTicker(root: string, expiry: string, type: "C" | "P", strike: number): string`
       - Per D-12: `{root}{YYMMDD}{C|P}{strike*1000 padded to 8 digits}`
       - expiry input: "YYYY-MM-DD" → extract YY, MM, DD
       - strike*1000, zero-padded to 8 digits: `String(Math.round(strike * 1000)).padStart(8, "0")`
       - Example: SPY, 2025-01-17, C, 470 → "SPY250117C00470000"

    3. `computeStrategyPnlPath(legs: ReplayLeg[], barsByLeg: MassiveBarRow[][]): PnlPoint[]`
       - Per D-06: mark price = HL2 = (high + low) / 2 at each minute
       - Per D-08: combined P&L = sum across legs of `(currentHL2 - entryPrice) * quantity * multiplier`
       - Align bars by timestamp across legs (only include timestamps where ALL legs have a bar)
       - Build PnlPoint[] with timestamp from bars (date + " " + time), strategyPnl, and legPrices array
       - Return empty array if any leg has no bars

    4. `computeReplayMfeMae(pnlPath: PnlPoint[]): { mfe: number; mae: number; mfeTimestamp: string; maeTimestamp: string }`
       - Per D-09: MFE = max(strategyPnl), MAE = min(strategyPnl)
       - Track timestamps where peaks occur
       - Empty path → { mfe: 0, mae: 0, mfeTimestamp: "", maeTimestamp: "" }
