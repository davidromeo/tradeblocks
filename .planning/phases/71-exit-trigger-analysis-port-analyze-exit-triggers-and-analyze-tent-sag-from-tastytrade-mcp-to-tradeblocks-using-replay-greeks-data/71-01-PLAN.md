    ```typescript
    export type TriggerType =
      | 'profitTarget' | 'stopLoss' | 'trailingStop'
      | 'dteExit' | 'ditExit' | 'clockTimeExit'
      | 'underlyingPriceMove' | 'positionDelta' | 'perLegDelta'
      | 'vixMove' | 'vix9dMove' | 'vix9dVixRatio'
      | 'slRatioThreshold' | 'slRatioMove';

    export interface ExitTriggerConfig {
      type: TriggerType;
      threshold: number;       // Meaning depends on type (dollars, %, delta, ratio, etc.)
      // Context-specific optional fields:
      expiry?: string;         // YYYY-MM-DD for dteExit
      openDate?: string;       // YYYY-MM-DD for ditExit
      clockTime?: string;      // "HH:MM" for clockTimeExit (threshold ignored)
      trailAmount?: number;    // Dollar trail for trailingStop (uses threshold if omitted)
      // Data maps for triggers that need external prices:
      underlyingPrices?: Map<string, number>;  // timestamp -> price
      vixPrices?: Map<string, number>;         // timestamp -> VIX price
      vix9dPrices?: Map<string, number>;       // timestamp -> VIX9D price
      // S/L ratio inputs:
      spreadWidth?: number;    // Width of spread in dollars (max loss = width * contracts * multiplier)
      contracts?: number;      // Number of contracts
      multiplier?: number;     // Default 100
    }

    export interface TriggerFireEvent {
      type: TriggerType;
      firedAt: string;         // Timestamp when trigger fired
      pnlAtFire: number;      // Strategy P&L when trigger fired
      index: number;           // Index into pnlPath
      detail?: string;         // Human-readable description
    }

    export interface ExitTriggerResult {
      triggers: TriggerFireEvent[];        // All triggers that fired (sorted by fire time)
      firstToFire: TriggerFireEvent | null; // Earliest trigger
      actualExit?: {
        timestamp: string;
        pnl: number;
        pnlDifference: number;            // firstToFire.pnl - actualExit.pnl
      };
      summary: string;                     // Human-readable summary
    }

    export interface LegGroupConfig {
      label: string;           // e.g., "put_spread", "call_spread"
      legIndices: number[];    // Indices into ReplayLeg[] / legPrices[]
      triggers: ExitTriggerConfig[];
    }

    export interface LegGroupResult {
      label: string;
      result: ExitTriggerResult;
      groupPnl: number[];      // P&L for this leg group at each timestamp
    }
    ```

    **Core functions:**

    1. `evaluateTrigger(trigger: ExitTriggerConfig, pnlPath: PnlPoint[], legs: ReplayLeg[]): TriggerFireEvent | null` — Evaluate a single trigger against the full P&L path. Return the first point where it fires, or null.

    2. `analyzeExitTriggers(config: { pnlPath: PnlPoint[]; legs: ReplayLeg[]; triggers: ExitTriggerConfig[]; actualExitTimestamp?: string; legGroups?: LegGroupConfig[]; }): { overall: ExitTriggerResult; legGroups?: LegGroupResult[]; }` — Run all triggers, find first-to-fire, compute actual exit comparison. For leg groups per D-06, compute per-group P&L from legPrices and evaluate per-group triggers.

    **Trigger evaluation details per D-04 (evaluate at every point):**

    - `profitTarget`: `point.strategyPnl >= trigger.threshold`
    - `stopLoss`: `point.strategyPnl <= -trigger.threshold`
    - `trailingStop`: Track running max P&L. Fire when `runningMax - point.strategyPnl >= (trigger.trailAmount ?? trigger.threshold)`
    - `dteExit`: Parse timestamp date, compute calendar days to `trigger.expiry`. Fire when `dte <= trigger.threshold`
    - `ditExit`: Compute calendar days from `trigger.openDate` to timestamp date. Fire when `dit >= trigger.threshold`
    - `clockTimeExit`: Extract time from timestamp. Fire when `time >= (trigger.clockTime ?? "15:00")`
    - `underlyingPriceMove`: Get underlying price at timestamp from `trigger.underlyingPrices`. Compute % move from first available price. Fire when `abs(pctMove) >= trigger.threshold`
    - `positionDelta`: Fire when `abs(point.netDelta ?? 0) >= trigger.threshold`
    - `perLegDelta`: Fire when any `abs(legGreeks[i].delta ?? 0) >= trigger.threshold`
    - `vixMove`: Get VIX at timestamp from `trigger.vixPrices`. Compute % move from first. Fire when `abs(pctMove) >= trigger.threshold`
    - `vix9dMove`: Same as vixMove but with `trigger.vix9dPrices`
    - `vix9dVixRatio`: Compute VIX9D/VIX ratio. Fire when `ratio >= trigger.threshold` (or `ratio <= trigger.threshold` if threshold < 1)
    - `slRatioThreshold`: Compute spread value as sum of abs(legPrice * quantity * multiplier) for short legs. Max loss = `trigger.spreadWidth * (trigger.contracts ?? 1) * (trigger.multiplier ?? 100)`. S/L ratio = spreadValue / maxLoss. Fire when `slRatio >= trigger.threshold`
    - `slRatioMove`: Compute S/L ratio at each point. Track initial S/L ratio (first point). Fire when `abs(currentSLRatio - initialSLRatio) >= trigger.threshold`

    **Per-group P&L computation for D-06:**
    For each leg group, compute group P&L at each timestamp: `sum over legIndices of (legPrices[i] - legs[i].entryPrice) * legs[i].quantity * legs[i].multiplier`

    **Actual exit comparison per D-05:**
    Find the pnlPath point closest to actualExitTimestamp. Compute `pnlDifference = firstToFire.pnlAtFire - actualExitPnl`.
